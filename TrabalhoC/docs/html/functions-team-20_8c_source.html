318int calcularMDC(int a, int b) {
319 int temp;
320 int maior;
321 while (b != 0) {
322 temp = b;
323 b = a % b;
324 a = temp;
325 maior = fabs(a); //fabs= módulo
326 }
327 return maior;
328}
